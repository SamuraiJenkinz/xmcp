    2. Add the catch-all route. This MUST be the LAST route registered in `create_app()`, after ALL other routes and after blueprint registration. Place it just before `logger.info("Flask app created successfully")`:
       ```python
       @app.route('/', defaults={'path': ''})
       @app.route('/<path:path>')
       def catch_all(path):
           """Serve React SPA when ATLAS_UI=react, otherwise fall through to classic UI."""
           # Guard: never capture API, auth, or existing functional routes
           _guarded = ('api/', 'auth/', 'chat/', 'login', 'logout')
           if any(path == p.rstrip('/') or path.startswith(p) for p in _guarded):
               return 'Not found', 404
           if app.config.get('ATLAS_UI') != 'react':
               # Classic mode: serve Jinja2 templates as before
               if not path:
                   return index()
               return redirect(url_for('index'))
           # React mode: serve static assets or fallback to index.html
           if path:
               asset = os.path.join(FRONTEND_DIST, path)
               if os.path.isfile(asset):
                   return send_from_directory(FRONTEND_DIST, path)
           return send_from_directory(FRONTEND_DIST, 'index.html')
       ```

The existing `@app.route("/chat")` route stays unchanged.
