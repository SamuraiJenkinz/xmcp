"""Token counting and context window management for gpt-4o-mini.

Uses tiktoken with the o200k_base encoding (gpt-4o-mini's tokenizer).
Prunes oldest non-system messages when the conversation approaches
the 128K context window limit.
"""

from __future__ import annotations

import logging

import tiktoken

logger = logging.getLogger(__name__)

_ENCODING = tiktoken.get_encoding("o200k_base")
_MAX_TOKENS = 128_000
_SAFETY_BUFFER = 4_096  # Reserve for response generation
_EFFECTIVE_LIMIT = _MAX_TOKENS - _SAFETY_BUFFER


def count_tokens_in_messages(messages: list[dict]) -> int:
    """Count tokens for a list of chat messages.

    Follows the OpenAI token counting cookbook for gpt-4o-mini:
    - 3 tokens reply primer
    - 3 tokens per-message overhead
    - Token count for each string value in the message
    - Token count for tool_call function name and arguments
    """
    tokens = 3  # Every reply is primed with 3 tokens
    for msg in messages:
        tokens += 3  # Per-message overhead
        for key, value in msg.items():
            if isinstance(value, str):
                tokens += len(_ENCODING.encode(value))
            elif key == "tool_calls" and isinstance(value, list):
                for tc in value:
                    fn = tc.get("function", {})
                    tokens += len(_ENCODING.encode(fn.get("name", "")))
                    tokens += len(_ENCODING.encode(fn.get("arguments", "")))
    return tokens


def prune_conversation(messages: list[dict]) -> list[dict]:
    """Remove oldest non-system messages until within token limit.

    Always preserves:
    - System messages (role == 'system')
    - The most recent user message (last in list with role == 'user')

    Removes from oldest to newest among non-system messages.

    Returns a new list (does not mutate the input).
    """
    system_msgs = [m for m in messages if m.get("role") == "system"]
    other_msgs = [m for m in messages if m.get("role") != "system"]

    total = count_tokens_in_messages(system_msgs + other_msgs)

    if total <= _EFFECTIVE_LIMIT:
        return messages  # No pruning needed

    pruned_count = 0
    while (
        count_tokens_in_messages(system_msgs + other_msgs) > _EFFECTIVE_LIMIT
        and len(other_msgs) > 1  # Always keep at least the latest message
    ):
        other_msgs.pop(0)  # Remove oldest non-system message
        pruned_count += 1

    if pruned_count > 0:
        logger.info(
            "Pruned %d messages to fit context window (%d tokens remaining)",
            pruned_count,
            count_tokens_in_messages(system_msgs + other_msgs),
        )

    return system_msgs + other_msgs


def get_token_count(messages: list[dict]) -> int:
    """Convenience wrapper for token counting."""
    return count_tokens_in_messages(messages)
```

"""Chat routes with SSE streaming for the Exchange assistant."""

from __future__ import annotations

import json
import logging

from flask import Blueprint, Response, request, session, stream_with_context

from chat_app.auth import login_required
from chat_app.context_mgr import prune_conversation
from chat_app.openai_client import (
    build_system_message,
    get_client,
    run_tool_loop,
    _message_to_dict,
    Config,
)

logger = logging.getLogger(__name__)

chat_bp = Blueprint("chat_routes", __name__)


@chat_bp.route("/chat/stream", methods=["POST"])
@login_required
def chat_stream():
    """SSE endpoint: run tool loop then stream final response.

    Expects JSON body: {"message": "user query text"}

    SSE event types:
    - {"type": "tool", "name": "tool_name", "status": "calling|success|error"}
    - {"type": "text", "content": "chunk of text"}
    - {"type": "done"}
    - {"type": "error", "message": "error description"}
    """
    data = request.get_json(silent=True) or {}
    user_message = data.get("message", "").strip()
    if not user_message:
        return Response(
            f"data: {json.dumps({'type': 'error', 'message': 'Empty message'})}\n\n",
            content_type="text/event-stream",
        )

    # --- Read session BEFORE entering generator ---
    # (Session is not accessible inside generator without stream_with_context,
    # and even with it, we should read what we need upfront)
    user = session.get("user", {})
    display_name = user.get("name", "Colleague")
    conversation = session.get("conversation", [])

    # Initialize conversation with system prompt if empty
    if not conversation:
        conversation = [build_system_message(display_name)]

    # Add user message
    conversation.append({"role": "user", "content": user_message})

    # Prune conversation before API call
    conversation = prune_conversation(conversation)

    def generate():
        nonlocal conversation
        try:
            # Phase 1: Run tool-calling loop (non-streaming)
            # Tool calls are fast (2-4s each), streaming partial JSON args is not useful
            conversation, tool_events = run_tool_loop(conversation)

            # Send tool status events
            for event in tool_events:
                yield f"data: {json.dumps({'type': 'tool', 'name': event['name'], 'status': event['status']})}\n\n"

            # The tool loop appended the final assistant message with content.
            # Now we want to stream the final response instead.
            # Remove the last assistant message (the non-streamed one) and re-request with streaming.
            last_msg = conversation[-1]
            if last_msg.get("role") == "assistant" and last_msg.get("content"):
                # The tool loop already got a complete response — stream it as chunks
                # This happens when tool loop completed and got a final answer
                full_text = last_msg["content"]
                # Remove and re-stream in chunks for UI effect
                conversation.pop()

            # Phase 2: Stream final AI response
            client = get_client()
            stream = client.chat.completions.create(
                model=Config.OPENAI_MODEL,
                messages=conversation,
                stream=True,
            )

            full_text = ""
            for chunk in stream:
                if chunk.choices and chunk.choices[0].delta:
                    delta = chunk.choices[0].delta
                    if delta.content:
                        full_text += delta.content
                        yield f"data: {json.dumps({'type': 'text', 'content': delta.content})}\n\n"

            # Append the complete assistant message to conversation
            conversation.append({"role": "assistant", "content": full_text})

            # Save updated conversation to session
            session["conversation"] = conversation

            # Phase 3: Done signal
            yield f"data: {json.dumps({'type': 'done'})}\n\n"

        except Exception as exc:
            logger.exception("Chat stream error")
            yield f"data: {json.dumps({'type': 'error', 'message': str(exc)})}\n\n"

    return Response(
        stream_with_context(generate()),
        content_type="text/event-stream",
        headers={
            "Cache-Control": "no-cache",
            "X-Accel-Buffering": "no",
            "Connection": "keep-alive",
        },
    )


@chat_bp.route("/chat/clear", methods=["POST"])
@login_required
def chat_clear():
    """Clear the current conversation."""
    session.pop("conversation", None)
    return {"status": "ok"}
```
