```python
_MAX_TOOL_ITERATIONS = 5  # Prevent infinite loops


def run_tool_loop(
    messages: list[dict],
    tools: list[dict] | None = None,
) -> tuple[list[dict], list[dict]]:
    """Execute the OpenAI tool-calling loop with MCP dispatch.

    Makes a chat completion with tool definitions. If the model returns
    tool_calls, dispatches each to MCP sequentially, appends results as
    tool-role messages, and calls again. Repeats until the model returns
    a text response (no tool_calls) or the iteration limit is reached.

    Args:
        messages: Conversation history (mutated in place with new messages).
        tools: OpenAI tool schemas. If None, uses cached MCP tools.

    Returns:
        Tuple of (updated_messages, tool_events) where tool_events is a
        list of dicts with tool name and status for UI display.
    """
    global _use_tools_param

    if tools is None:
        tools = get_openai_tools()

    client = get_client()
    tool_events: list[dict] = []

    for iteration in range(_MAX_TOOL_ITERATIONS):
        # Build API call kwargs
        kwargs = {
            "model": Config.OPENAI_MODEL,
            "messages": messages,
        }

        if tools and _use_tools_param:
            kwargs["tools"] = tools
            kwargs["tool_choice"] = "auto"
        elif tools and not _use_tools_param:
            # Fallback: deprecated 'functions' parameter for older API versions
            kwargs["functions"] = [t["function"] for t in tools]
            kwargs["function_call"] = "auto"

        try:
            response = client.chat.completions.create(**kwargs)
        except Exception as exc:
            error_str = str(exc)
            # Detect if gateway rejects 'tools' parameter
            if _use_tools_param and (
                "tools" in error_str.lower()
                or "unrecognized" in error_str.lower()
            ):
                logger.warning(
                    "Gateway rejected 'tools' parameter, falling back to 'functions': %s",
                    error_str,
                )
                _use_tools_param = False
                # Retry with 'functions' parameter
                kwargs.pop("tools", None)
                kwargs.pop("tool_choice", None)
                kwargs["functions"] = [t["function"] for t in tools]
                kwargs["function_call"] = "auto"
                response = client.chat.completions.create(**kwargs)
            else:
                raise

        choice = response.choices[0]
        assistant_msg = choice.message

        # Convert SDK message to plain dict immediately
        assistant_dict = _message_to_dict(assistant_msg)
        messages.append(assistant_dict)

        # Check for tool calls (handle both 'tools' and 'functions' style)
        tool_calls = assistant_msg.tool_calls
        function_call = getattr(assistant_msg, "function_call", None)

        if tool_calls:
            # Modern 'tools' style — dispatch each sequentially
            for tc in tool_calls:
                tool_name = tc.function.name
                tool_args_str = tc.function.arguments
                tool_events.append({"name": tool_name, "status": "calling"})
                logger.info("Tool call: %s(%s)", tool_name, tool_args_str)

                try:
                    tool_args = json.loads(tool_args_str) if tool_args_str else {}
                    result = call_mcp_tool(tool_name, tool_args)
                    tool_events[-1]["status"] = "success"
                except Exception as tool_exc:
                    result = f"Error calling {tool_name}: {tool_exc}"
                    tool_events[-1]["status"] = "error"
                    logger.error("Tool call failed: %s — %s", tool_name, tool_exc)

                messages.append({
                    "tool_call_id": tc.id,
                    "role": "tool",
                    "name": tool_name,
                    "content": result,
                })

        elif function_call:
            # Deprecated 'functions' style fallback
            tool_name = function_call.name
            tool_args_str = function_call.arguments
            tool_events.append({"name": tool_name, "status": "calling"})
            logger.info("Function call (legacy): %s(%s)", tool_name, tool_args_str)

            try:
                tool_args = json.loads(tool_args_str) if tool_args_str else {}
                result = call_mcp_tool(tool_name, tool_args)
                tool_events[-1]["status"] = "success"
            except Exception as tool_exc:
                result = f"Error calling {tool_name}: {tool_exc}"
                tool_events[-1]["status"] = "error"
                logger.error("Function call failed: %s — %s", tool_name, tool_exc)

            messages.append({
                "role": "function",
                "name": tool_name,
                "content": result,
            })

        else:
            # No tool calls — model produced a final text response
            break

    return messages, tool_events
```
