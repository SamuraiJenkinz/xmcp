import asyncio
import logging
import threading
from contextlib import AsyncExitStack

from mcp import ClientSession, StdioServerParameters
from mcp.client.stdio import stdio_client

logger = logging.getLogger(__name__)

_mcp_session: ClientSession | None = None
_mcp_loop: asyncio.AbstractEventLoop | None = None
_mcp_tools: list[dict] | None = None  # Cached OpenAI-format tool schemas
_mcp_lock = threading.Lock()  # Serialize concurrent MCP calls
_exit_stack: AsyncExitStack | None = None  # Kept alive for subprocess lifetime

**Background event loop:**
```python
def _start_mcp_event_loop() -> None:
    """Run a dedicated asyncio event loop in a daemon thread.

    This loop runs forever and processes all MCP async operations.
    Never call asyncio.run() from Flask request handlers — dispatch
    coroutines to this loop via _async_run instead.
    """
    global _mcp_loop
    _mcp_loop = asyncio.new_event_loop()
    asyncio.set_event_loop(_mcp_loop)
    _mcp_loop.run_forever()


def _async_run(coro, timeout: float = 60.0):
    """Execute a coroutine on the MCP background event loop.

    Called from synchronous Flask request handlers. Blocks until
    the coroutine completes or times out.
    """
    if _mcp_loop is None:
        raise RuntimeError("MCP event loop not started")
    future = asyncio.run_coroutine_threadsafe(coro, _mcp_loop)
    return future.result(timeout=timeout)
```

**Connection setup:**
```python
async def _connect_mcp() -> None:
    """Connect to the MCP server subprocess and cache tool schemas."""
    global _mcp_session, _mcp_tools, _exit_stack

    _exit_stack = AsyncExitStack()
    server_params = StdioServerParameters(
        command="uv",
        args=["run", "python", "-m", "exchange_mcp.server"],
        env=None,  # Inherit environment from Flask process
    )

    transport = await _exit_stack.enter_async_context(
        stdio_client(server_params)
    )
    read_stream, write_stream = transport

    session = await _exit_stack.enter_async_context(
        ClientSession(read_stream, write_stream)
    )
    await session.initialize()

    # Enumerate tools and convert to OpenAI function schema format
    response = await session.list_tools()
    _mcp_tools = [
        {
            "type": "function",
            "function": {
                "name": tool.name,
                "description": (tool.description or "")[:1024],
                "parameters": tool.inputSchema,
            },
        }
        for tool in response.tools
    ]

    _mcp_session = session
    logger.info(
        "MCP client connected: %d tools available", len(_mcp_tools)
    )
```

**Public API:**
```python
def init_mcp(app=None) -> None:
    """Initialize MCP client — spawn server subprocess and cache tools.

    Called once during Flask app startup. Starts a daemon thread with
    a dedicated asyncio event loop, then connects to the MCP server.
    """
    thread = threading.Thread(target=_start_mcp_event_loop, daemon=True)
    thread.start()

    # Brief wait for the event loop to initialize
    import time
    time.sleep(0.2)

    try:
        _async_run(_connect_mcp(), timeout=30.0)
    except Exception:
        logger.exception("Failed to connect to MCP server")
        raise


def get_openai_tools() -> list[dict]:
    """Return MCP tools as OpenAI function-calling schemas.

    These are cached at startup — no async call needed.
    """
    return _mcp_tools or []


def call_mcp_tool(name: str, arguments: dict) -> str:
    """Call an MCP tool synchronously from a Flask request handler.

    Dispatches the async call to the background event loop and blocks
    until the result is available. A threading lock ensures sequential
    access to the MCP stdio session (which does not support concurrent
    JSON-RPC calls).

    Args:
        name: Tool name (e.g., 'get_mailbox_stats').
        arguments: Tool arguments as a dict.

    Returns:
        Tool result as a string (from TextContent).

    Raises:
        RuntimeError: If MCP session is not connected or tool call fails.
    """
    if _mcp_session is None:
        raise RuntimeError("MCP session not connected")

    async def _call():
        result = await _mcp_session.call_tool(name, arguments)
        if result.content:
            return result.content[0].text
        return ""

    with _mcp_lock:
        return _async_run(_call(), timeout=60.0)


def is_connected() -> bool:
    """Check if the MCP client is connected."""
    return _mcp_session is not None and _mcp_tools is not None
```
