**Create `frontend/src/utils/syntaxHighlightJson.ts`:**

Zero-dependency regex JSON highlighter. Takes a `JSON.stringify` output string and wraps tokens in span elements.

```typescript
/**
 * Syntax-highlight a JSON string by wrapping tokens in <span> elements.
 * ONLY pass output of JSON.stringify — never raw user input (XSS risk).
 */
export function syntaxHighlightJson(json: string): string {
  return json.replace(
    /("(\\u[a-zA-Z0-9]{4}|\\[^u]|[^\\"])*"(\s*:)?|\b(true|false|null)\b|-?\d+(?:\.\d*)?(?:[eE][+-]?\d+)?)/g,
    (match) => {
      let cls = 'json-number';
      if (/^"/.test(match)) {
        cls = /:$/.test(match) ? 'json-key' : 'json-string';
      } else if (/true|false/.test(match)) {
        cls = 'json-bool';
      } else if (/null/.test(match)) {
        cls = 'json-null';
      }
      return `<span class="${cls}">${match}</span>`;
    }
  );
}
```

function formatElapsed(ms: number): string {
  if (ms < 1000) return `${ms}ms`;
  return `${(ms / 1000).toFixed(1)}s`;
}

Update `prettyResult` to return both plain text and highlighted HTML:
```typescript
function prettyJson(result: string): { plain: string; highlighted: string } {
  try {
    const parsed = JSON.stringify(JSON.parse(result), null, 2);
    return { plain: parsed, highlighted: syntaxHighlightJson(parsed) };
  } catch {
    return { plain: result, highlighted: result };
  }
}
```

Replace the component with the redesigned version:

```tsx
export function ToolPanel({ name, status, params, result, startTime, endTime }: Props) {
  const [isOpen, setIsOpen] = useState(false);
  const hasParams = Object.keys(params).length > 0;

  const elapsedMs = (startTime != null && endTime != null)
    ? Math.round((endTime - startTime) * 1000)
    : null;

  const statusLabel = status === 'success' ? 'Done' : 'Error';

  // Pre-compute JSON for both params and result
  const paramsJson = hasParams ? syntaxHighlightJson(JSON.stringify(params, null, 2)) : '';
  const resultData = result !== null ? prettyJson(result) : null;

  return (
    <details
      className="tool-panel"
      onToggle={(e) => setIsOpen((e.currentTarget as HTMLDetailsElement).open)}
    >
      <summary className="tool-panel-summary">
        <ChevronRight20Regular
          className={`tool-panel-chevron${isOpen ? ' tool-panel-chevron-open' : ''}`}
        />
        <span className="tool-panel-name">{name}</span>
        <span className={`tool-panel-badge tool-panel-badge-${status}`}>
          {statusLabel}
        </span>
        {elapsedMs !== null && (
          <span className="tool-panel-elapsed">Ran in {formatElapsed(elapsedMs)}</span>
        )}
      </summary>
      <div className="tool-panel-body">
        {hasParams && (
          <>
            <div className="tool-panel-label">Parameters</div>
            <pre
              className="tool-panel-json"
              dangerouslySetInnerHTML={{ __html: paramsJson }}
            />
          </>
        )}
        {resultData !== null && (
          <>
            <div className="tool-panel-label">Exchange Result</div>
            <pre
              className="tool-panel-json tool-panel-result"
              dangerouslySetInnerHTML={{ __html: resultData.highlighted }}
            />
            <CopyButton getText={() => resultData.plain} />
          </>
        )}
      </div>
    </details>
  );
}
```

Key decisions:
- Use `onToggle` event on `<details>` to sync `isOpen` state (avoids chevron getting stuck — research pitfall #1).
- `startTime`/`endTime` are optional; elapsed only renders when both present (historical messages won't have them — research pitfall #5).
- JSON.parse then JSON.stringify before dangerouslySetInnerHTML for XSS safety (research pitfall #4).
- CopyButton getText returns the PLAIN text, not the highlighted HTML.
- Replace old `.tool-panel-icon` (8px dot) with ChevronRight20Regular.
- Replace old `.tool-panel-status` with `.tool-panel-badge` for clearer semantics.
