    ```python
    """End-to-end Exchange Online connectivity verification.

    Requires: AZURE_CERT_THUMBPRINT, AZURE_CLIENT_ID, AZURE_TENANT_DOMAIN env vars set.
    Requires: ExchangeOnlineManagement PowerShell module installed.
    Requires: Certificate installed in CurrentUser store.
    """
    import asyncio
    import json
    import sys
    import os

    # Add project root to path
    sys.path.insert(0, os.path.dirname(os.path.dirname(os.path.abspath(__file__))))
    from exchange_mcp.exchange_client import ExchangeClient

    async def main():
        print("=" * 60)
        print("Exchange Online Connectivity Verification")
        print("=" * 60)

        # Step 1: Verify env vars
        print("\n[1/4] Checking environment variables...")
        client = ExchangeClient(timeout=30)
        print("  OK: All required env vars present")

        # Step 2: Test connection with verify_connection()
        print("\n[2/4] Testing Exchange Online connection (Get-OrganizationConfig)...")
        connected = await client.verify_connection()
        if not connected:
            print("  FAIL: verify_connection() returned False")
            print("  Check: Certificate installed? App permissions granted? Module installed?")
            sys.exit(1)
        print("  OK: Connected to Exchange Online")

        # Step 3: Run proof-of-concept cmdlet
        print("\n[3/4] Running proof-of-concept cmdlet (Get-OrganizationConfig full)...")
        result = await client.run_cmdlet(
            "Get-OrganizationConfig | Select-Object Name, DisplayName, Identity, WhenCreated | ConvertTo-Json -Depth 10"
        )
        # Note: ConvertTo-Json is in the cmdlet_line here, but _build_cmdlet_script also adds it.
        # The implementation should handle this gracefully (double ConvertTo-Json is a no-op on a string).
        # ALTERNATIVELY: just pass the cmdlet without ConvertTo-Json and let the template handle it.
        # CORRECTION: The _build_cmdlet_script template already adds ConvertTo-Json. So the cmdlet_line
        # should be JUST the cmdlet + Select-Object, NOT including ConvertTo-Json.

        # Re-do with correct pattern:
        result = await client.run_cmdlet(
            "Get-OrganizationConfig | Select-Object Name, DisplayName, Identity, WhenCreated"
        )
        print(f"  Result: {json.dumps(result, indent=2, default=str)}")

        # Verify expected fields are present
        expected_fields = ["Name", "DisplayName"]
        for field in expected_fields:
            if field not in result:
                print(f"  WARN: Expected field '{field}' not found in result")
            else:
                print(f"  OK: {field} = {result[field]}")

        # Step 4: Verify session cleanup (no orphaned connections)
        print("\n[4/4] Verifying session cleanup...")
        # Run a second cmdlet to prove no session limit issues
        result2 = await client.run_cmdlet(
            "Get-OrganizationConfig | Select-Object Name"
        )
        print(f"  OK: Second call succeeded (no orphaned sessions)")

        print("\n" + "=" * 60)
        print("ALL CHECKS PASSED - Exchange Online client is working")
        print("=" * 60)

    if __name__ == "__main__":
        asyncio.run(main())
    ```

    **scripts/verify_dns.py** — DNS lookup proof:

    ```python
    """End-to-end DNS resolver verification for DMARC/SPF lookups."""
    import asyncio
    import json
    import sys
    import os

    sys.path.insert(0, os.path.dirname(os.path.dirname(os.path.abspath(__file__))))
    from exchange_mcp.dns_utils import get_dmarc_record, get_spf_record

    async def main():
        print("=" * 60)
        print("DNS Resolver Verification (DMARC/SPF)")
        print("=" * 60)

        test_domain = "google.com"  # Known to have both DMARC and SPF

        print(f"\n[1/2] Looking up DMARC record for {test_domain}...")
        dmarc = await get_dmarc_record(test_domain)
        print(f"  Result: {json.dumps(dmarc, indent=2)}")
        assert dmarc["found"], f"DMARC not found for {test_domain}"
        print(f"  OK: DMARC policy = {dmarc.get('policy', 'N/A')}")

        print(f"\n[2/2] Looking up SPF record for {test_domain}...")
        spf = await get_spf_record(test_domain)
        print(f"  Result: {json.dumps(spf, indent=2)}")
        assert spf["found"], f"SPF not found for {test_domain}"
        print(f"  OK: SPF mechanisms = {len(spf.get('mechanisms', []))} entries")

        print("\n" + "=" * 60)
        print("ALL DNS CHECKS PASSED")
        print("=" * 60)

    if __name__ == "__main__":
        asyncio.run(main())
    ```
